\section{汇编语言的位操作指令}
\begin{center}
    写出汇编语言例句
\end{center}

\subsection{逻辑运算(1分)}
\begin{minted}{asm}
    and %eax,%ebx
    or  %eax,%ebx
    xor %eax,%ebx
    not %eax
\end{minted}

\subsection{无符号数左右移（2分）}
\begin{minted}{asm}
    mov $4,%cl
    shl %cl,%eax
    shr $4,%ebx
\end{minted}

\subsection{有符号左右移（2分）}
\begin{minted}{asm}
    mov $4,%cl
    sal %cl,%eax
    sar $4,%ebx
\end{minted}

\subsection{循环移位（2分）}
\begin{minted}{asm}
    rol 1,%eax
    ror 2,%ebx
\end{minted}

\subsection{带进位位的循环移位（2分）}
\begin{minted}{asm}
    rcl 1,%eax
    rcr 2,%ebx
\end{minted}

\subsection{测试、位测试BTx（2分）}
\begin{minted}{asm}
    testb $1,%al
    testw $2.%ax
    testl $4,%eax
    testq $8,%rax
\end{minted}

\subsection{条件传送CMOVxx（2分）}
\begin{minted}{asm}
    cmove   %eax,%ebx
    cmoves  %eax,%ebx
    cmoveng %eax,%ebx
    cmova   %eax,%ebx
\end{minted}

\subsection{条件设置SETCxx（1分）}
\begin{minted}{asm}
    sete  %al
    sets  %bl
    setge %cl
\end{minted}

\subsection{进位位操作（1分）}
\begin{minted}{asm}
    add %eax,%ebx
    adc %ecx,%edx
\end{minted}
